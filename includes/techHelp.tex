\begin{lstlisting}[
  caption={advancedTraining.py},
  label={lst:python},
  language=python,
  style=code-listing
]
from ultralytics import YOLO
import pandas as pd
import os


# =====================================================
# RESULTS STORAGE
# =====================================================

results_summary = []

# =====================================================
# DATASET PATHS
# =====================================================

DEFAULT_DATASET = "cesta k /YOLODataset/dataset.yaml"

BINARY_DATASET = "cesta k /YOLODataset_binary/dataset.yaml"

# =====================================================
# TRAINING FUNCTION
# =====================================================

def run_training(
    model_name,
    dataset_path,
    run_name,
    augmentation=False
):

    print("\n====================================")
    print(f"Starting: {run_name}")
    print("====================================\n")

    # load model
    model = YOLO(model_name)

    # =============================================
    # TRAIN
    # =============================================

    metrics = model.train(

        # dataset
        data=dataset_path,

        # max epochs
        epochs=300,

        # EARLY STOPPING
        patience=50,

        # image size
        imgsz=640,

        # batch size
        batch=8,

        # =========================================
        # LIGHT DOCUMENT AUGMENTATION
        # =========================================

        hsv_h=0.01 if augmentation else 0.0,

        hsv_s=0.3 if augmentation else 0.0,

        hsv_v=0.2 if augmentation else 0.0,

        degrees=3 if augmentation else 0.0,

        translate=0.05 if augmentation else 0.0,

        scale=0.15 if augmentation else 0.0,

        shear=2 if augmentation else 0.0,

        # disable flips for documents
        fliplr=0.0,
        flipud=0.0,

        # IMPORTANT FOR DOCUMENTS
        mosaic=0.0 if augmentation else 1.0,

        # =========================================
        # SAVE
        # =========================================

        project="/yolo_training",

        name=run_name
    )

    # =============================================
    # READ RESULTS CSV
    # =============================================

    results_path = f"/yolo_training/{run_name}/results.csv"

    results_csv = pd.read_csv(results_path)

    completed_epochs = len(results_csv)

    # =============================================
    # SAVE SUMMARY
    # =============================================

    results_summary.append({

        "Run": run_name,

        "Dataset": dataset_path.split("/")[-2],

        "Augmentation": augmentation,

        "Max Epochs": 300,

        "Completed Epochs": completed_epochs,

        "Patience": 50,

        "mAP50": metrics.results_dict.get("metrics/mAP50(B)", None),

        "mAP50-95": metrics.results_dict.get("metrics/mAP50-95(B)", None),

        "Precision": metrics.results_dict.get("metrics/precision(B)", None),

        "Recall": metrics.results_dict.get("metrics/recall(B)", None)
    })

# =====================================================
# DEFAULT DATASET
# =====================================================

run_training(

    model_name="yolo11s.pt",

    dataset_path=DEFAULT_DATASET,

    run_name="yolo11s_default_earlystop_p50",

    augmentation=False
)

# =====================================================
# BINARY DATASET
# =====================================================

run_training(

    model_name="yolo11s.pt",

    dataset_path=BINARY_DATASET,

    run_name="yolo11s_binary_earlystop_p50",

    augmentation=False
)

# =====================================================
# AUGMENTED DATASET
# =====================================================

run_training(

    model_name="yolo11s.pt",

    dataset_path=DEFAULT_DATASET,

    run_name="yolo11s_augmented_lightaug_p50",

    augmentation=True
)

# =====================================================
# CREATE SUMMARY TABLE
# =====================================================

df = pd.DataFrame(results_summary)

# =====================================================
# SAVE CSV
# =====================================================

csv_path = "cesta k csv"

df.to_csv(csv_path, index=False)

# =====================================================
# PRINT SUMMARY
# =====================================================

print("\n====================================")
print("ALL TRAININGS FINISHED")
print("====================================")

print("\nSUMMARY:")
print(df)

print("\nCSV saved to:")
print(csv_path)

\paragraph{Použitie natrénovaných váh YOLO}:

Príprava súborov:
\begin{itemize}
    \item Umiestnite natrénované váhy (napr. best.pt) do zvoleného adresára.
    \item Pripravte si adresár s obrázkami, ktoré chcete analyzovať.
\end{itemize}
Spustenie detekčného skriptu:
\begin{lstlisting}[
  caption={plottingVal.py},
  label={lst:python},
  language=python,
  style=code-listing
]
import os
import cv2
import numpy as np
from ultralytics import YOLO

MODEL_PATH = "cesta/k/best.pt"
INPUT_FOLDER = "cesta/k/obrazkom"
OUTPUT_FOLDER = "cesta/pre/vystup"

os.makedirs(OUTPUT_FOLDER, exist_ok=True)
model = YOLO(MODEL_PATH)

for filename in os.listdir(INPUT_FOLDER):
    if filename.lower().endswith((".jpg", ".jpeg", ".png")):
        input_path = os.path.join(INPUT_FOLDER, filename)
        output_path = os.path.join(OUTPUT_FOLDER, filename)
        frame = cv2.imread(input_path)
        results = model.predict(frame, save=False)
        img = results[0].plot(font_size=60, pil=True)
        img_np = np.array(img)
        img_bgr = cv2.cvtColor(img_np, cv2.COLOR_RGB2BGR)
        cv2.imwrite(output_path, img_bgr)
        print(f"Processed: {filename}")
